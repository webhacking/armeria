49      private static final boolean USE_EPOLL = getBoolean("useEpoll", Epoll.isAvailable(),

72      private static final long DEFAULT_SERVER_IDLE_TIMEOUT_MILLIS =
73              getLong("defaultServerIdleTimeoutMillis",

86      private static final String DEFAULT_DEFAULT_BACKOFF_SPEC =
87              "exponential=200:10000,jitter=0.2,maxAttempts=10";
88      private static final String DEFAULT_BACKOFF_SPEC =
89              getNormalized("defaultBackoffSpec", DEFAULT_DEFAULT_BACKOFF_SPEC, value -> {
90                  try {
91                      Backoff.of(value);
92                      return true;
93                  } catch (Exception e) {
94                      // Invalid backoff specification
95                      return false;

99      static {
100         if (!Epoll.isAvailable()) {
101             final Throwable cause = filterCause(Epoll.unavailabilityCause());
102             logger.info("/dev/epoll not available: {}", cause.toString());
103         } else if (USE_EPOLL) {
104             logger.info("Using /dev/epoll");
105         }
106 
107         if (!OpenSsl.isAvailable()) {
108             final Throwable cause = filterCause(OpenSsl.unavailabilityCause());
109             logger.info("OpenSSL not available: {}", cause.toString());
110         } else if (USE_OPENSSL) {
111             logger.info("Using OpenSSL: {}, 0x{}",
112                         OpenSsl.versionString(),
113                         Long.toHexString(OpenSsl.version() & 0xFFFFFFFFL));
114         }
115     }

242     private static boolean getBoolean(String name, boolean defaultValue) {
243         return getBoolean(name, defaultValue, value -> true);
244     }
245 
246     private static boolean getBoolean(String name, boolean defaultValue, Predicate<Boolean> validator) {
247         return "true".equals(getNormalized(name, String.valueOf(defaultValue), value -> {
248             if ("true".equals(value)) {
249                 return validator.test(true);
250             }
251 
252             if ("false".equals(value)) {
253                 return validator.test(false);
254             }
255 
256             return false;
257         }));
258     }
259 
260     private static int getInt(String name, int defaultValue, IntPredicate validator) {
261         return Integer.parseInt(getNormalized(name, String.valueOf(defaultValue), value -> {
262             try {
263                 return validator.test(Integer.parseInt(value));
264             } catch (Exception e) {
265                 // null or non-integer
266                 return false;
267             }
268         }));
269     }
270 
271     private static long getLong(String name, long defaultValue, LongPredicate validator) {
272         return Long.parseLong(getNormalized(name, String.valueOf(defaultValue), value -> {
273             try {
274                 return validator.test(Long.parseLong(value));
275             } catch (Exception e) {
276                 // null or non-integer
277                 return false;
278             }
279         }));
280     }
281 
282     private static String getNormalized(String name, String defaultValue, Predicate<String> validator) {
283         final String fullName = PREFIX + name;
284         final String value = getLowerCased(fullName);
285         if (value == null) {
286             logger.info("{}: {} (default)", fullName, defaultValue);
287             return defaultValue;
288         }
289 
290         if (validator.test(value)) {
291             logger.info("{}: {}", fullName, value);
292             return value;
293         }
294 
295         logger.info("{}: {} (default instead of: {})", fullName, defaultValue, value);
296         return defaultValue;
297     }
298 
299     private static String getLowerCased(String fullName) {
300         String value = System.getProperty(fullName);
301         if (value != null) {
302             value = Ascii.toLowerCase(value);
303         }
304         return value;
305     }
306 
307     private static Throwable filterCause(Throwable cause) {
308         if (cause instanceof ExceptionInInitializerError) {
309             return cause.getCause();
310         }
311 
312         return cause;
313     }
