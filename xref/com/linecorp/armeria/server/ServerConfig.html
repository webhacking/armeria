71      ServerConfig(
72              Iterable<ServerPort> ports,
73              VirtualHost defaultVirtualHost, Iterable<VirtualHost> virtualHosts,
74              EventLoopGroup workerGroup, boolean shutdownWorkerGroupOnStop,
75              int maxNumConnections, long idleTimeoutMillis,
76              long defaultRequestTimeoutMillis, long defaultMaxRequestLength,
77              Duration gracefulShutdownQuietPeriod, Duration gracefulShutdownTimeout,
78              Executor blockingTaskExecutor, String serviceLoggerPrefix) {
79  
80          requireNonNull(ports, "ports");
81          requireNonNull(virtualHosts, "virtualHosts");
82          requireNonNull(defaultVirtualHost, "defaultVirtualHost");
83  
84          // Set the primitive properties.
85          this.workerGroup = requireNonNull(workerGroup, "workerGroup");
86          this.shutdownWorkerGroupOnStop = shutdownWorkerGroupOnStop;
87          this.maxNumConnections = validateMaxNumConnections(maxNumConnections);
88          this.idleTimeoutMillis = validateIdleTimeoutMillis(idleTimeoutMillis);
89          this.defaultRequestTimeoutMillis = validateDefaultRequestTimeoutMillis(defaultRequestTimeoutMillis);
90          this.defaultMaxRequestLength = validateDefaultMaxRequestLength(defaultMaxRequestLength);
91          this.gracefulShutdownQuietPeriod = validateNonNegative(requireNonNull(
92                  gracefulShutdownQuietPeriod), "gracefulShutdownQuietPeriod");
93          this.gracefulShutdownTimeout = validateNonNegative(requireNonNull(
94                  gracefulShutdownTimeout), "gracefulShutdownTimeout");
95          validateGreaterThanOrEqual(gracefulShutdownTimeout, "gracefulShutdownTimeout",
96                                     gracefulShutdownQuietPeriod, "gracefulShutdownQuietPeriod");
97  
98          requireNonNull(blockingTaskExecutor, "blockingTaskExecutor");
99          if (blockingTaskExecutor instanceof ExecutorService) {
100             this.blockingTaskExecutor = new InterminableExecutorService((ExecutorService) blockingTaskExecutor);
101         } else {
102             this.blockingTaskExecutor = new ExecutorBasedExecutorService(blockingTaskExecutor);
103         }
104 
105         this.serviceLoggerPrefix = ServiceConfig.validateLoggerName(serviceLoggerPrefix, "serviceLoggerPrefix");
106 
107         // Set localAddresses.
108         final List<ServerPort> portsCopy = new ArrayList<>();
109         for (ServerPort p : ports) {
110             if (p == null) {
111                 break;
112             }
113             portsCopy.add(p);
114         }
115 
116         if (portsCopy.isEmpty()) {
117             throw new IllegalArgumentException("no ports in the server");
118         }
119 
120         this.ports = Collections.unmodifiableList(portsCopy);
121 
122         // Set virtual host definitions and initialize their domain name mapping.
123         defaultVirtualHost = normalizeDefaultVirtualHost(defaultVirtualHost, portsCopy);
124         final DomainNameMappingBuilder<VirtualHost> mappingBuilder =
125                 new DomainNameMappingBuilder<>(defaultVirtualHost);
126         final List<VirtualHost> virtualHostsCopy = new ArrayList<>();
127         for (VirtualHost h : virtualHosts) {
128             if (h == null) {
129                 break;
130             }
131             virtualHostsCopy.add(h);
132             mappingBuilder.add(h.hostnamePattern(), h);
133         }
134         virtualHostMapping = mappingBuilder.build();
135 
136         // Add the default VirtualHost to the virtualHosts so that a user can retrieve all VirtualHosts
137         // via virtualHosts(). i.e. no need to check defaultVirtualHost().
138         virtualHostsCopy.add(defaultVirtualHost);
139 
140         // Sets the parent of VirtualHost to this configuration.
141         virtualHostsCopy.forEach(h -> h.setServerConfig(this));
142 
143         if (virtualHostsCopy.stream().allMatch(h -> h.serviceConfigs().isEmpty())) {
144             throw new IllegalArgumentException("no services in the server");
145         }
146 
147         this.virtualHosts = Collections.unmodifiableList(virtualHostsCopy);
148         this.defaultVirtualHost = defaultVirtualHost;
149 
150         // Build the complete list of the services available in this server.
151         services = virtualHostsCopy.stream()
152                                    .flatMap(h -> h.serviceConfigs().stream())
153                                    .collect(toImmutableList());
154     }

160     static long validateIdleTimeoutMillis(long idleTimeoutMillis) {
161         if (idleTimeoutMillis < 0) {
162             throw new IllegalArgumentException("idleTimeoutMillis: " + idleTimeoutMillis + " (expected: >= 0)");
163         }
164         return idleTimeoutMillis;
165     }

167     static long validateDefaultRequestTimeoutMillis(long defaultRequestTimeoutMillis) {
168         if (defaultRequestTimeoutMillis < 0) {
169             throw new IllegalArgumentException(
170                     "defaultRequestTimeoutMillis: " + defaultRequestTimeoutMillis + " (expected: >= 0)");
171         }
172         return defaultRequestTimeoutMillis;
173     }

175     static long validateDefaultMaxRequestLength(long defaultMaxRequestLength) {
176         if (defaultMaxRequestLength < 0) {
177             throw new IllegalArgumentException(
178                     "defaultMaxRequestLength: " + defaultMaxRequestLength + " (expected: >= 0)");
179         }
180         return defaultMaxRequestLength;

183     static Duration validateNonNegative(Duration duration, String fieldName) {
184         if (duration.isNegative()) {
185             throw new IllegalArgumentException(fieldName + ": " + duration + " (expected: >= 0)");
186         }
187         return duration;
188     }
189 
190     static void validateGreaterThanOrEqual(Duration larger, String largerFieldName,
191                                            Duration smaller, String smallerFieldName) {
192         if (larger.compareTo(smaller) < 0) {
193             throw new IllegalArgumentException(largerFieldName + " must be greater than or equal to" +
194                                                smallerFieldName);
195         }
196     }

198     private static VirtualHost normalizeDefaultVirtualHost(VirtualHost h, List<ServerPort> ports) {
199         final SslContext sslCtx = h.sslContext();
200         // The default virtual host must have sslContext set if TLS is in use.
201         if (sslCtx == null && ports.stream().anyMatch(p -> p.protocol().isTls())) {
202             throw new IllegalArgumentException(
203                     "defaultVirtualHost must have sslContext set when TLS is enabled.");
204         }
205 
206         return new VirtualHost(
207                 h.defaultHostname(), "*", sslCtx,
208                 h.serviceConfigs().stream().map(
209                         e -> new ServiceConfig(e.pathMapping(), e.service(), e.loggerName().orElse(null)))
210                  .collect(Collectors.toList()));
211     }

224     void setServer(Server server) {
225         if (this.server != null) {
226             throw new IllegalStateException("ServerConfig cannot be used for more than one Server.");

228         this.server = requireNonNull(server, "server");

268     /**
269      * Finds the {@link List} of {@link VirtualHost}s that contains the specified {@link Service}. If there's
270      * no match, an empty {@link List} is returned. Note that this is potentially an expensive operation and
271      * thus should not be used in a performance-sensitive path.
272      */
273     public List<VirtualHost> findVirtualHosts(Service<?, ?> service) {
274         requireNonNull(service, "service");
275 
276         @SuppressWarnings("rawtypes")
277         final Class<? extends Service> serviceType = service.getClass();
278         final List<VirtualHost> res = new ArrayList<>();
279         for (VirtualHost h : virtualHosts) {
280             for (ServiceConfig c : h.serviceConfigs()) {
281                 // Consider the case where the specified service is decorated before being added.
282                 final Service<?, ?> s = c.service();
283                 @SuppressWarnings("rawtypes")
284                 Optional<? extends Service> sOpt = s.as(serviceType);
285                 if (!sOpt.isPresent()) {
286                     continue;
287                 }
288 
289                 if (sOpt.get() == service) {
290                     res.add(c.virtualHost());
291                     break;
292                 }
293             }
294         }
295 
296         return res;
297     }

383     @Override
384     public String toString() {
385         String strVal = this.strVal;
386         if (strVal == null) {
387             this.strVal = strVal = toString(
388                     getClass(), ports(), null, virtualHosts(),
389                     workerGroup(), shutdownWorkerGroupOnStop(),
390                     maxNumConnections(), idleTimeoutMillis(),
391                     defaultRequestTimeoutMillis(), defaultMaxRequestLength(),
392                     gracefulShutdownQuietPeriod(), gracefulShutdownTimeout(),
393                     blockingTaskExecutor(), serviceLoggerPrefix());
394         }
395 
396         return strVal;
397     }

399     static String toString(
400             Class<?> type,
401             Iterable<ServerPort> ports, VirtualHost defaultVirtualHost, List<VirtualHost> virtualHosts,
402             EventLoopGroup workerGroup, boolean shutdownWorkerGroupOnStop,
403             int maxNumConnections, long idleTimeoutMillis,
404             long defaultRequestTimeoutMillis, long defaultMaxRequestLength,
405             Duration gracefulShutdownQuietPeriod, Duration gracefulShutdownTimeout,
406             Executor blockingTaskExecutor, String serviceLoggerPrefix) {
407 
408         StringBuilder buf = new StringBuilder();
409         if (type != null) {
410             buf.append(type.getSimpleName());
411         }
412 
413         buf.append("(ports: [");
414 
415         boolean hasPorts = false;
416         for (ServerPort p : ports) {
417             buf.append(ServerPort.toString(null, p.localAddress(), p.protocol()));
418             buf.append(", ");
419             hasPorts = true;
420         }
421 
422         if (hasPorts) {
423             buf.setCharAt(buf.length() - 2, ']');
424             buf.setCharAt(buf.length() - 1, ',');
425         } else {
426             buf.append("],");
427         }
428 
429         buf.append(" virtualHosts: [");
430         if (!virtualHosts.isEmpty()) {
431             virtualHosts.forEach(c -> {
432                 buf.append(VirtualHost.toString(null, c.defaultHostname(), c.hostnamePattern(),
433                                                 c.sslContext(), c.serviceConfigs()));
434                 buf.append(", ");
435             });
436 
437             if (defaultVirtualHost != null) {
438                 buf.append(VirtualHost.toString(null, defaultVirtualHost.defaultHostname(), "*",
439                                                 defaultVirtualHost.sslContext(),
440                                                 defaultVirtualHost.serviceConfigs()));
441             } else {
442                 buf.setLength(buf.length() - 2);
443             }
444         } else if (defaultVirtualHost != null) {
445             buf.append(VirtualHost.toString(null, defaultVirtualHost.defaultHostname(), "*",
446                                             defaultVirtualHost.sslContext(),
447                                             defaultVirtualHost.serviceConfigs()));
448         }
449 
450         buf.append("], workerGroup: ");
451         buf.append(workerGroup);
452         buf.append(" (shutdownOnStop=");
453         buf.append(shutdownWorkerGroupOnStop);
454         buf.append("), maxNumConnections: ");
455         buf.append(maxNumConnections);
456         buf.append(", idleTimeout: ");
457         buf.append(idleTimeoutMillis);
458         buf.append("ms, defaultRequestTimeout: ");
459         buf.append(defaultRequestTimeoutMillis);
460         buf.append("ms, defaultMaxRequestLength: ");
461         buf.append(defaultMaxRequestLength);
462         buf.append("B, gracefulShutdownQuietPeriod: ");
463         buf.append(gracefulShutdownQuietPeriod);
464         buf.append(", gracefulShutdownTimeout: ");
465         buf.append(gracefulShutdownTimeout);
466         buf.append(", blockingTaskExecutor: ");
467         buf.append(blockingTaskExecutor);
468         buf.append(", serviceLoggerPrefix: ");
469         buf.append(serviceLoggerPrefix);
470         buf.append(')');
471 
472         return buf.toString();
473     }
