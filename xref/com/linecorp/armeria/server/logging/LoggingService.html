17  package com.linecorp.armeria.server.logging;
18  
19  import static java.util.Objects.requireNonNull;
20  
21  import java.util.function.Function;
22  
23  import org.slf4j.Logger;
24  
25  import com.google.common.annotations.VisibleForTesting;
26  
27  import com.linecorp.armeria.common.HttpHeaders;
28  import com.linecorp.armeria.common.Request;
29  import com.linecorp.armeria.common.Response;
30  import com.linecorp.armeria.common.logging.LogLevel;
31  import com.linecorp.armeria.common.logging.RequestLog;
32  import com.linecorp.armeria.common.logging.RequestLogAvailability;
33  import com.linecorp.armeria.server.Service;
34  import com.linecorp.armeria.server.ServiceRequestContext;
35  import com.linecorp.armeria.server.SimpleDecoratingService;
36  
37  /**
38   * Decorates a {@link Service} to log {@link Request}s and {@link Response}s.
39   *
40   * @param <I> the {@link Request} type
41   * @param <O> the {@link Response} type
42   */
43  public final class LoggingService<I extends Request, O extends Response> extends SimpleDecoratingService<I, O> {
44  
45      @VisibleForTesting
46      static final String REQUEST_FORMAT = "Request: {}";
47      @VisibleForTesting
48      static final String RESPONSE_FORMAT = "Response: {}";
49  
50      /**
51       * Returns a new {@link Service} decorator that logs {@link Request}s and {@link Response}s at
52       * {@link LogLevel#INFO} for success, {@link LogLevel#WARN} for failure.
53       *
54       * @see LoggingServiceBuilder for more information on the default settings.
55       */
56      public static <I extends Request, O extends Response>
57      Function<Service<I, O>, LoggingService<I, O>> newDecorator() {
58          return new LoggingServiceBuilder()
59                  .requestLogLevel(LogLevel.INFO)
60                  .successfulResponseLogLevel(LogLevel.INFO)
61                  .failureResponseLogLevel(LogLevel.WARN)
62                  .newDecorator();
63      }
64  
65      /**
66       * @deprecated Use {@link LoggingServiceBuilder}.
67       */
68      @Deprecated
69      public static <I extends Request, O extends Response>
70      Function<Service<I, O>, LoggingService<I, O>> newDecorator(LogLevel level) {
71          return delegate -> new LoggingService<>(delegate, level);
72      }
73  
74      private final LogLevel requestLogLevel;
75      private final LogLevel successfulResponseLogLevel;
76      private final LogLevel failedResponseLogLevel;
77      private final Function<HttpHeaders, HttpHeaders> requestHeadersSanitizer;
78      private final Function<Object, Object> requestContentSanitizer;
79      private final Function<HttpHeaders, HttpHeaders> responseHeadersSanitizer;
80      private final Function<Object, Object> responseContentSanitizer;
81      private final Sampler sampler;
82  
83      /**
84       * @deprecated Use {@link LoggingService#newDecorator()}.
85       */
86      @Deprecated
87      public LoggingService(Service<I, O> delegate) {
88          this(delegate, LogLevel.INFO);
89      }
90  
91      /**
92       * @deprecated Use {@link LoggingServiceBuilder}.
93       */
94      @Deprecated
95      public LoggingService(Service<I, O> delegate, LogLevel level) {
96          this(delegate,
97               level,
98               level,
99               level,
100              Function.identity(),
101              Function.identity(),
102              Function.identity(),
103              Function.identity(),
104              Sampler.ALWAYS_SAMPLE);
105     }
106 
107     /**
108      * Creates a new instance that logs {@link Request}s and {@link Response}s at the specified
109      * {@link LogLevel}s with the specified sanitizers.
110      */
111     LoggingService(
112             Service<I, O> delegate,
113             LogLevel requestLogLevel,
114             LogLevel successfulResponseLogLevel,
115             LogLevel failedResponseLogLevel,
116             Function<HttpHeaders, HttpHeaders> requestHeadersSanitizer,
117             Function<Object, Object> requestContentSanitizer,
118             Function<HttpHeaders, HttpHeaders> responseHeadersSanitizer,
119             Function<Object, Object> responseContentSanitizer,
120             Sampler sampler) {
121         super(requireNonNull(delegate, "delegate"));
122         this.requestLogLevel = requireNonNull(requestLogLevel, "requestLogLevel");
123         this.successfulResponseLogLevel =
124                 requireNonNull(successfulResponseLogLevel, "successfulResponseLogLevel");
125         this.failedResponseLogLevel = requireNonNull(failedResponseLogLevel, "failedResponseLogLevel");
126         this.requestHeadersSanitizer = requireNonNull(requestHeadersSanitizer, "requestHeadersSanitizer");
127         this.requestContentSanitizer = requireNonNull(requestContentSanitizer, "requestContentSanitizer");
128         this.responseHeadersSanitizer = requireNonNull(responseHeadersSanitizer, "responseHeadersSanitizer");
129         this.responseContentSanitizer = requireNonNull(responseContentSanitizer, "resposneContentSanitizer");
130         this.sampler = requireNonNull(sampler, "sampler");
131     }
132 
133     @Override
134     public O serve(ServiceRequestContext ctx, I req) throws Exception {
135         if (sampler.isSampled()) {
136             ctx.log().addListener(this::logRequest, RequestLogAvailability.REQUEST_END);
137             ctx.log().addListener(this::logResponse, RequestLogAvailability.COMPLETE);
138         }
139         return delegate().serve(ctx, req);
140     }
141 
142     /**
143      * Logs a stringified request of {@link RequestLog}.
144      */
145     private void logRequest(RequestLog log) {
146         final Logger logger = ((ServiceRequestContext) log.context()).logger();
147         if (requestLogLevel.isEnabled(logger)) {
148             requestLogLevel.log(logger, REQUEST_FORMAT,
149                                 log.toStringRequestOnly(requestHeadersSanitizer, requestContentSanitizer));
150         }
151     }
152 
153     /**
154      * Logs a stringified response of {@link RequestLog}.
155      */
156     private void logResponse(RequestLog log) {
157         final Logger logger = ((ServiceRequestContext) log.context()).logger();
158         final LogLevel level =
159                 log.responseCause() == null ? successfulResponseLogLevel : failedResponseLogLevel;
160         if (level.isEnabled(logger)) {
161             level.log(logger, RESPONSE_FORMAT,
162                       log.toStringResponseOnly(responseHeadersSanitizer, responseContentSanitizer));
163         }
164     }
165 }
